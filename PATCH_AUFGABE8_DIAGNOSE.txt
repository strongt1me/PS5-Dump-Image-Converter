"""
PATCH FÜR AUFGABE 8: Bessere Fehlerbehandlung beim Dump Validator
================================================================

Problem:
- Wenn eine .ffpfsc-Datei unvollständig extrahiert wird (nur pfs_image.dat),
  zeigt der Validator eine unspezifische Fehlermeldung
- Benutzer wissen nicht, warum kritische Dateien fehlen

Lösung:
- Prüfe auf "Unvollständige Extraktion" wenn eboot.bin fehlt
- Zeige präzise Diagnose-Tipps statt generischer Fehlermeldung
- Log bei kritischen Fehlern zusätzliche Debug-Info

Implementierung:
-  Neue Konstante CRITICAL_FILES mit eboot.bin, sce_sys/param.json, etc.
- Zusätzliche Prüfung in dump_validator.py nach den erforderlichen Ordnern
- Spezifische Fehler-Ausgabe bei fehlenden kritischen Dateien
"""

# Beispiel-Output VORHER (generisch):
BEFORE = """
[ERGEBNIS] FEHLGESCHLAGEN: SchnellPruefung fehlgeschlagen: eboot.bin nicht gefunden
"""

# Beispiel-Output NACHHER (spezifisch):
AFTER = """
[ERGEBNIS] FEHLGESCHLAGEN: Dump unvollständig
[DIAGNOSE] Kritische Dateien fehlen: eboot.bin, sce_sys/param.json, sce_sys/pfs-version.dat
           Dies deutet auf eine unvollständig extrahierte .ffpfsc-Datei hin.
           
[TIPPS]    - SHA256-Prüfsumme der .ffpfsc-Datei vergleichen
           - MkPFS-Version aktualisieren
           - Dump mit mehr freiem Speicherplatz neu erstellen
"""

# INTEGRATION in die Hauptdatei (PS5ImageConverter_Pro_FINAL_revised.py):
# Wenn Aufgabe 8 einen Dump validiert und die kritischen Dateien fehlen,
# soll die console_view diese detailliertere Meldung anzeigen.

# Betroffene Methode: _execute_dump_validator()
# In dieser Methode wird nach der Validierung das Ergebnis geparsed.
# Code-Snippet für bessere Fehlerbehandlung:

CODE_SNIPPET = '''
# Nach Validator-Ausführung:
if result.status == "FAILED":
    critical_missing = result.summary.get("critical_missing", [])
    if critical_missing:
        # Unvollständige Extraktion erkannt
        msg = (
            f"\\n[DIAGNOSE] Kritische Dateien fehlen: {', '.join(critical_missing)}\\n"
            f"[HINWEIS]  Die .ffpfsc-Datei wurde möglicherweise nicht vollständig extrahiert.\\n"
            f"[TIPPS]    - SHA256-Checksum der .ffpfsc-Datei überprüfen\\n"
            f"           - MkPFS-Version aktualisieren\\n"
            f"           - Mit mehr freiem Speicherplatz neu versuchen\\n"
        )
        self._append_to_log(msg)
'''
